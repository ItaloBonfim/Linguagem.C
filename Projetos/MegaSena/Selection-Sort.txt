#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <time.h>
#include <string.h>


void selection_sort(int vet[], int tam){
	int i,j,temp, aux;
	
	   for(i=0;i<(tam-1);i++){
	   	for(j=i+1;j<tam;j++){
		   aux=i;  
	     if(vet[aux]>vet[j]){          
	   	  	   aux=j;
	   	  	   			 }                                               
	   	  	   			 
	   	  	   if(aux != i){
	   	  	   	      temp=vet[i];
	   	  	   	      vet[i]=vet[aux];
	   	  	   	      vet[aux]=temp;
					}			 
							}
	   } 	   
}


int main(){
 
 int num=10, x;
  int vetor[10]={5,1,3,9,10,22,11,26,2,8};
      
	   selection_sort(vetor,num);
	   
	   for(x=0;x<num;x++){
	   	
	   	printf("Vetor Ordenado= %d\n", vetor[x]);
	   }
	   

	   
	   system("pause");
	   return 0;
	
}
